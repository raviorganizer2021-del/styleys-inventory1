-- Run this FULL SQL in Supabase SQL Editor CREATE TABLE IF NOT EXISTS items ( id bigserial primary key, sku text, item_code text, name text, category text, qty numeric default 0, threshold numeric default 10, unit text default 'pcs', price numeric default 0, hsn text, location text, set_size numeric default 1, monthly_sales text default '{}', created_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS sales ( id bigserial primary key, order_no text, customer text, customer_gstin text, sale_date date, subtotal numeric, gst_rate numeric default 5, gst numeric, total numeric, payment_status text default 'Pending', color text default '', created_by text default '', created_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS sale_items ( id bigserial primary key, sale_id bigint references sales(id) on delete cascade, item_id bigint, item_name text, sku text, qty numeric, price numeric, unit text, amount numeric ); CREATE TABLE IF NOT EXISTS challans ( id bigserial primary key, challan_no text, customer text, customer_address text, challan_date date, purpose text, remarks text, color text default '', created_by text default '', created_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS challan_items ( id bigserial primary key, challan_id bigint references challans(id) on delete cascade, item_id bigint, item_name text, sku text, qty numeric, unit text, remarks text ); CREATE TABLE IF NOT EXISTS china_stock ( id bigserial primary key, sku text unique, item_name text, qty numeric default 0, monthly_sales text default '{}', created_at timestamptz default now(), updated_at timestamptz default now() ); CREATE TABLE IF NOT EXISTS china_plan ( id bigserial primary key, sku text, item_name text, supplier text, qty_needed numeric default 0, estimated_cost numeric default 0, delivery_date date, status text default 'Pending', created_at timestamptz default now() );
