CREATE TABLE items ( id bigserial primary key, sku text, name text, category text, qty numeric default 0, threshold numeric default 10, unit text default 'pcs', price numeric default 0, hsn text, location text, created_at timestamptz default now() ); CREATE TABLE sales ( id bigserial primary key, order_no text, customer text, customer_gstin text, sale_date date, subtotal numeric, gst_rate numeric default 5, gst numeric, total numeric, payment_status text default 'Pending', created_at timestamptz default now() ); CREATE TABLE sale_items ( id bigserial primary key, sale_id bigint references sales(id) on delete cascade, item_id bigint, item_name text, sku text, qty numeric, price numeric, unit text, amount numeric ); CREATE TABLE challans ( id bigserial primary key, challan_no text, customer text, customer_address text, challan_date date, purpose text, remarks text, created_at timestamptz default now() ); CREATE TABLE challan_items ( id bigserial primary key, challan_id bigint references challans(id) on delete cascade, item_id bigint, item_name text, sku text, qty numeric, unit text, remarks text ); ALTER TABLE items ENABLE ROW LEVEL SECURITY; ALTER TABLE sales ENABLE ROW LEVEL SECURITY; ALTER TABLE sale_items ENABLE ROW LEVEL SECURITY; ALTER TABLE challans ENABLE ROW LEVEL SECURITY; ALTER TABLE challan_items ENABLE ROW LEVEL SECURITY; CREATE POLICY "all" ON items FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "all" ON sales FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "all" ON sale_items FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "all" ON challans FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "all" ON challan_items FOR ALL USING (true) WITH CHECK (true);
